=== TRIGGERS ===
Trigger: trg_printing_items_updated_at
  Table: printing_items
  SQL: CREATE TRIGGER trg_printing_items_updated_at
AFTER UPDATE ON printing_items
FOR EACH ROW
BEGIN
    UPDATE printing_items
    SET updated_at = CURRENT_TIMESTAMP
    WHERE id = OLD.id;
END
